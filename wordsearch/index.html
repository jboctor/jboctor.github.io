                    function isPresent(, searchGrid) {
                        if (word == '') {
                            return true;
                        }
                        var letter = word.substr(0, 1);
                        word = word.substr(1);
                        for (var i in searchGrid) {
                            for (var j in searchGrid[i]) {
                                if (searchGrid[i][j] == letter) {
                                    if (this.isPresent(word, this.createNewSearchGrid(parseInt(i), parseInt(j)))) {
                                        return true;
                                    }
                                }
                            }
                        }
                        return false;
                    }
